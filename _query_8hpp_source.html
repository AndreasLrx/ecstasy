519 template <typename F>
520 void splitThreads(std::size_t batchSize, F &&fct, bool wait = true)
521 {
522 Iterator end = this->end();
523 Iterator current = this->begin();
524 Iterator currentBatchStart = current;
525 size_t currentBatchSize = 0;
526 std::vector<std::thread> threads;
527
528 while (current != end) {
529 ++currentBatchSize;
530 // Find the next element (which may be the end sentinel)
531 ++current;
532 // Batch full, we start a new thread and reset the current batch informations
533 if (currentBatchSize == batchSize) {
534 threads.emplace_back(processBatch<F>, currentBatchStart, current, fct);
535 currentBatchSize = 0;
536 currentBatchStart = current;
537 }
538 }
539
540 if (currentBatchSize > 0)
541 threads.emplace_back(processBatch<F>, currentBatchStart, current, fct);
542
543 if (wait)
544 for (auto &thread : threads)
545 thread.join();
546 }
