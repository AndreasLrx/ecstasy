349 QueryImplementation(First &first, Others &...others) : _storages(first, others...)
350 {
352 if constexpr (is_queryable_with_adjust_v<First>
353 || std::disjunction_v<is_queryable_with_adjust<Others>...>) {
354 size_t maxSize = std::max({getQueryableMask(first).size(), getQueryableMask(others).size()...});
355
356 adjustMask(first, maxSize);
357 (adjustMask(others, maxSize), ...);
358 }
359 this->_mask = (util::BitSet(getQueryableMask(first)) &= ... &= getQueryableMask(others));

443 template <typename F>
444 void splitThreads(std::size_t batchSize, F &&fct, bool wait = true)
445 {
446 Iterator end = this->end();
447 Iterator current = this->begin();
448 Iterator currentBatchStart = current;
449 size_t currentBatchSize = 0;
450 std::vector<std::thread> threads;
451
452 while (current != end) {
453 ++currentBatchSize;
454 // Find the next element (which may be the end sentinel)
455 ++current;
456 // Batch full, we start a new thread and reset the current batch informations
457 if (currentBatchSize == batchSize) {
458 threads.emplace_back(processBatch<F>, currentBatchStart, current, fct);
459 currentBatchSize = 0;
460 currentBatchStart = current;
461 }
462 }
463
464 if (currentBatchSize > 0)
465 threads.emplace_back(processBatch<F>, currentBatchStart, current, fct);
466
467 if (wait)
468 for (auto &thread : threads)
469 thread.join();
470 }
