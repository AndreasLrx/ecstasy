326 S &updateEntity(RegistryEntity &entity)

328 std::size_t component_hash = loadComponentHash();
329
330 if (this->hasEntityComponentSerializer(component_hash)) {
331 IEntityComponentSerializer &component = this->getEntityComponentSerializer(component_hash);
332 IStorage &storage = entity.getRegistry().getStorages().get(component.getStorageTypeIndex());
