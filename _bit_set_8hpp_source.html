229 BIT_SET_CONSTEXPR std::size_t firstSet(std::size_t start = 0) const
230 {
231 std::uint64_t mask = (~std::uint64_t(0)) << (start & 0b111111);
232 std::size_t word_pos = start >> 6;
233 std::uint64_t word = this->_store[word_pos] & mask;
234
235 while (word == 0) {
236 ++word_pos;
237 word = this->_store[word_pos];
238 }
239 return (word_pos << 6) + static_cast<std::size_t>(std::countr_zero(word));
240 }

249 BIT_SET_CONSTEXPR std::size_t firstUnset(std::size_t start = 0) const
250 {
251 std::uint64_t mask = (~std::uint64_t(0)) << (start & 0b111111);
252 std::size_t word_pos = start >> 6;
253 std::uint64_t word = this->_store[word_pos] | (~mask);
254
255 while (word == (~std::uint64_t(0))) {
256 ++word_pos;
257 word = this->_store[word_pos];
258 }
259 return (word_pos << 6) + static_cast<std::size_t>(std::countr_one(word));
260 }
